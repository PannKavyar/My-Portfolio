            const name = "Aung Zin Latt";

            const name = "Aung Zin Latt";

            const sayBye = function() {

            sayBye();

            const name = "Aung Zin Latt";

Writing Arrow Function with Arguments

အပေါ်က ကျွန်တော်တို့ရေးခဲ့တဲ့ arrow function မှာတော့ arguments တွေမပါပါဘူး။ တစ်ကယ်လို့ arguments တွေပါခဲ့ရင် သူတို့ ဘာတွေအလုပ်လုပ်လဲဆိုတာကို လေ့လာကြည့်ရအောင်။ ပထမဆုံး single argument ပါတဲ့ arrow function နှစ်ခုကိုတည်ဆောက်လိုက်မယ်။ ပြီးရင်တော့ အပေါ်ကရေးခဲ့သလိုပဲ variable တစ်ခုကြေငြာပြီး သူ့ထဲကို (x) => {} ဆိုတဲ့ function ကို assign လုပ်ပေးရမှာဖြစ်ပါတယ်။ အောက်က example ကို တစ်ချက်ကြည့်ရအောင်။

Arrow function with single argument:

        
            const square = (x) => {
                return x * x;
            }
            square(3);

            const cube = (x) => {
                return square(x) * x;
            }
            cube(5);

Arrow function with multiple arguments:

        
            const multiply = (x, y) => {
                return x * y;
            }
            multiply(4, 5);

            const add = (a, b) => {
                return a + b;
            }
            add(10, 20);

            const subtract = (a, b) => {
                return a - b;
            }
            subtract(50, 20);

Concise Arrow Function Syntax

Arrow functions တွေဟာ function declarations တွေထက်ပိုပြီး ရေးရတာလွယ်သလို ပိုပြီးလည်း တိုတိုလေးနဲ့ ရိုးရှင်းပါတယ်။ ဒါပေမယ့် arrow functions တွေကတော့ ပိုပြီး ကျစ်လစ်ပါတယ်။ တစ်ကယ်လို့ arrow function က single argument တစ်ခုကိုပဲ ယူမယ်ဆိုရင် parenthesis ထည့်ပေးဖို့ မလိုပါဘူး။ တစ်ကယ်လို့ ကျွန်တော်တို့ရေးတဲ့ function က argument မပါတာ (သို့) multiple arguments တွေနဲ့ဆိုရင်တော့ parenthesis ကို ဖြုတ်ပစ်လို့မရပါဘူး။ Single argument ပဲပါဝင်တဲ့ functions တွေသာလျှင် parenthesis ဖြုတ်ပြီး ရေးလို့ရတာဖြစ်ပါတယ်။ တစ်ကယ်လို့ function ထဲမှာ line of code တစ်ကြောင်းတည်းရေးချင်တယ်ဆိုရင် return ဆိုတဲ့ keyword ကိုတောင် ထည့်ပေးဖို့မလိုပါဘူး။ သူဟာ return keyword မပါရင်တောင် statement ရဲ့ value ကို အလိုအလျှောက် return ပြန်ပေးမှာဖြစ်ပါတယ်။ ဒီထက်မကဘဲ arrow function ထဲမှာ single line of code တစ်ကြောင်းတည်းပဲရေးမယ်ဆိုရင် curly braces ကိုပါ ဖြုတ်ပြီး reduce လုပ်ပစ်လို့ရပါတယ်။ ကျွန်တော်တို့ function ထဲမှာ code တစ်ကြောင်းဘဲ return ပြန်ဖို့ရှိတယ်ဆိုရင် ဒီလိုရေးတာဟာ ကျွန်တော်တို့ code ကို ပို clear ဖြစ်ပြီး ပိုမိုကျစ်လစ်စေမှာဖြစ်ပါတယ်။ Arrow functions တွေကို မရင်းနှီးသေးရင်တော့ ဒါဟာ ရှုပ်ထွေးတယ်ထင်ရပေမယ့် ဒါကိုနားလည်သွားရင်တော့ ကျွန်တော်တို့ code က တစ်ကယ်ကို standard ကျပြီး ပိုမိုရှင်းလင်းနေမှာဖြစ်ပါတယ်။ Modern browser အားလုံးမှာလည်း Arrow Function Syntax ကို support လုပ်ပေးထားပါတယ်။

If Else Statement in JavaScript

ဒီတစ်ခါကျွန်တော်တို့က JavaScript မှာ decision making တွေပြုလုပ်တဲ့အခါ လိုအပ်တဲ့ conditions တွေအကြောင်းကို လေ့လာသွားမှာဖြစ်ပါတယ်။ ပထမဆုံးတစ်ခုကတော့ if else statement ပါ။ သူ့ကို ဘာကြောင့် conditional လို့ခေါ်လဲဆိုရင် condition တစ်ခုက true ဖြစ်နေပြီး တစ်ခုက false ဖြစ်နေလို့ပဲဖြစ်ပါတယ်။ if, else statement က follow လုပ်တဲ့ paths တွေကိုတော့ branches လို့ခေါ်ပါတယ်။ သူတို့ဟာ directions အမျိုးမျိုးမှာရှိတဲ့ program logic ရဲ့ flow ကို branch အဖြစ် ခွဲထုတ်လိုက်တာဖြစ်ပါတယ်။ ပြောရရင် မှန်ရင်ဒါကိုလုပ် ၊ မှားရင် အခြားဟာကို လုပ်ဆိုတာမျိုးပါ။ အဲ့ဒီ if else statement မှာ two way conditional နဲ့ multi way conditional ဆိုပြီးနှစ်မျိုးရှိပါတယ်။ ပထမတစ်ခုက if ကမှန်ရင် တစ်ခုလုပ်မယ်။ ပြီးမှ if က မှားသွားရင် အခြားတစ်ခုကိုလုပ်မယ်ဆိုပြီး နှစ်ခုတည်းစစ်တာဖြစ်ပါတယ်။ ဒုတိယတစ်ခုကတော့ တစ်ခုမှန်လားစစ်တယ်၊ ပြီးရင်နောက်တစ်ခုမှန်လားထပ်စစ်တယ်။ အဲ့လိုနဲ့ ကျွန်တော်တို့ စစ်ချင်တာတွေကို အကုန်လိုက်စစ်လိုက်ပြီးမှ မှန်တဲ့ကောင်ကိုပဲ အလုပ်လုပ်သွားတာဖြစ်ပါတယ်။ ဒါကို ကျွန်တော်တို့က else if condition လို့လည်း ခေါ်ကြပါတယ်။ Example ကိုအောက်မှာကြည့်ရအောင်။

Example of Two way conditional:

        
            var isTrue = true;  // false

            if (isTrue) {
                console.log("Yes");
            } else {
                console.log("No");
            }

Example of Multi way conditional:

        
            var day = 8;

            if (day === 0) {
                console.log('Today is Sunday');
            } else if (day === 1) {
                console.log('Today is Monday');
            } else if (day === 2) {
                console.log('Today is Tuesday');
            } else if (day === 3) {
                console.log('Today is Wednesday');
            } else if (day === 4) {
                console.log('Today is Thursday');
            } else if (day === 5) {
                console.log('Today is Friday');
            } else if (day === 6) {
                console.log('Today is Saturday');
            } else {
                console.log('Invalid Day!');
            }

Switch Statement

Switch statement ကို ကျွန်တော်တို့က multiple branching logic တွေအတွက် အသုံးပြုပါတယ်။ Switch statement ဟာ if statement နဲ့လည်း ဆင်တူပါတယ်။ သူ့မှာ parentheses ပါမယ်။ သူ့နောက်မှာ block of code ကို hold လုပ်ပေးတဲ့ curly braces ပါမယ်။ Parentheses တွေကတော့ expression တစ်ခုကို hold လုပ်ပါတယ်။ ကျွန်တော်တို့ switch ထဲက parenthesis ထဲမှာ variable ဖြစ်တဲ့ day ကိုထည့်လိုက်မယ်။ ပြီးတော့ if statement နဲ့မတူတာတစ်ခုက cases အားလုံးဟာ only one block of code အတွင်းမှာပဲရှိနေတာဖြစ်ပါတယ်။ condition တစ်ခုချင်းစီကိုတော့ case ဆိုတဲ့ keyword နဲ့ mark လုပ်ပါတယ်။ သူ့ရဲ့အလုပ်လုပ်ပုံကတော့ JavaScript interpreter ကို switch ထဲက variable ဟာ case ထဲက value နဲ့ ညီရင် သူ့ထဲက block code ကို execute လုပ်ပေးတာဖြစ်ပါတယ်။ case တစ်ခုပြီးရင်တော့ အမြဲတမ်း break လုပ်ပေးရပါတယ်။ တစ်ကယ်လို့ break statement ကိုရောက်သွားပြီးဆိုရင်တော့ block code တစ်ခုလုံးကို jump လုပ်ပြီး နောက် instructions တွေကို ဆက်ပြီးအလုပ်လုပ်ပေးပါတယ်။ ကျွန်တော်တို့စစ်မယ့် condition အားလုံးမမှန်တော့ဘူးဆိုရင်တော့ switch statement က provide လုပ်ပေးထားတဲ့ default ဆိုတဲ့ option ကို အလုပ်လုပ်သွားမှာဖြစ်ပါတယ်။

For Example:

        
            var day = 1;

            switch (day) {
                case 0:
                    console.log('Sunday');
                    break;
                case 1:
                    console.log('Monday');
                    break;
                case 2:
                    console.log('Tuesday');
                    break;
                case 3:
                    console.log('Wednesday');
                    break;
                case 4:
                    console.log('Thursday');
                    break;
                case 5:
                    console.log('Friday');
                    break;
                case 6:
                    console.log('Saturday');
                    break;
                default:
                    console.log('Invalid Day!');
                    break;

Ternary Operator

Ternary operator ဟာ တစ်ကယ်ကို compact ဖြစ်တဲ့ operator တစ်ခုဖြစ်ပြီး programmer အများစုဟာ တစ်ကယ် concise ဖြစ်တဲ့ code တွေရေးဖို့အတွက် အသုံးပြုတဲ့ two-way branch တစ်ခုလည်းဖြစ်ပါတယ်။ သူ့ရဲ့ problem ကတော့ အလွန် compact ဖြစ်တဲ့အတွက် read လုပ်ဖို့ တစ်ကယ်ကို ခက်ခဲပါတယ်။ သူ့မှာ expression သုံးခုပါတဲ့အတွက်ကြောင့် ternary လို့ခေါ်တာဖြစ်ပါတယ်။ ပထမဆုံး expression က နောက်လာမယ့် expression နှစ်ခုထဲက ဘယ်ဟာကို run ပါမယ်ဆိုတာကို ဆုံးဖြတ်ပေးတဲ့ boolean တစ်ခုဖြစ်ပါတယ်။ Ternary operator ဟာာ colon (:) နဲ့ question mark (?) ဆိုပြီး character နှစ်ခုကို အသုံးပြုပါတယ်။ အဲ့ဒီ character နှစ်ခုဟာ ကျွန်တော်တို့ condition အတွက် markers တွေဖြစ်ပြီး branches နှစ်ခုလည်းဖြစ်ပါတယ်။ For example: <boolean> ? <expression if true> : <expression if false> ။ Ternary operator ရဲ့အလုပ်လုပ်ပုံကတော့ condition က true ဖြစ်ရင် question mark (?) နောက်က code ကို evaluate လုပ်ပေးပြီး၊ false ဖြစ်နေရင်တော့ colon (:) နောက်က code ကို evaluate လုပ်ပေးတာဖြစ်ပါတယ်။ နောက်ထပ် ရေးနည်းတစ်ခုကတော့ expression တစ်ခုအဖြစ် wild မှာ ပေါ်လာမယ့် ternary pop up တစ်ခုဖြစ်ပါတယ်။ JavaScript မှာဆိုရင် expression ဟာ value တစ်ခုကို resolves လုပ်တဲ့အရာဖြစ်ပါတယ်။ Expression တစ်ခုကို variable တစ်ခုထဲမှာ assign လုပ်နိုင်ပြီး အဲ့ဒီ variable ထဲမှာလည်း value ပါဝင်ပါလိမ့်မယ်။ ဥပမာ var x = 2 + 5; ဆိုရင် 2 +5 က expression တစ်ခုဖြစ်ပြီး 7 ဆိုတဲ့ result ရအောင် resolve လုပ်သွားတာဖြစ်ပါတယ်။ x ဟာ 2 + 5 ဆိုတဲ့ expression ကို hold မလုပ်ဘဲ value ဖြစ်တဲ့ 7 ကိုပဲ hold လုပ်ပြီး return ပြန်ပေးတာဖြစ်ပါတယ်။

Boolean Operators

AND (&&) ၊ OR (||) စတဲ့ boolean operators တွေဟာ multiple conditions တွေကို statement တစ်ကြောင်းတည်းနဲ့ testing လုပ်တဲ့နေရာမှာ တစ်ကယ်ကို အသုံးဝင်တဲ့ operators တွေဖြစ်ပါတယ်။ AND operator လိုပဲ condition တစ်ခုလုံး အလုပ်မလုပ်ခင် သူ့ထဲက condition အားလုံး true ဖြစ်မှသာ အလုပ်လုပ်ပေးတာဖြစ်ပါတယ်။ Programmer အချို့ကတော့ Short Circuit တွေလုပ်ဆောင်ဖို့အတွက် boolean operators တွေကို အသုံးပြုကြပါတယ်။ ဒါဟာ expression တစ်ခု evaluate လုပ်တာကို stop လုပ်လိုက်ခြင်းဖြင့် code ကို optimize လုပ်တဲ့ နည်းလမ်းတစ်ခုဖြစ်ပါတယ်။ သူ့ကို value တစ်ခုကို မြန်ဆန်စွာ assign လုပ်တာမျိုး (သို့) code ကို execute လုပ်တဲ့အခါမျိုးမှာ အသုံးပြုပါတယ်။ Boolean operators တွေကို စစ်တဲ့အခါမှာ JavaScript Interpreter က leftmost အဖြစ်ဆုံး expression ကနေစပြီး right ဘက်ကို တစ်ခုချင်း စစ်သွားတာဖြစ်ပါတယ်။ တစ်ကယ်လို့ အဲ့ဒီ operands တွေထဲက တစ်ခုဖြစ်ဖြစ် falsey ဖြစ်နေမယ်ဆိုရင်တော့ ပထမဆုံး falsey ဖြစ်တဲ့ operand ကိုပဲ return ပြန်ပေးမှာဖြစ်ပါတယ်။ ECMAScript 2015 မှာ default parameter တွေ function ထဲကို short circuiting သုံးပြီးတော့ ပြုလုပ်တာက အသုံးနည်းခဲ့ပါတယ်။ 2015 အသစ်မှာတော့ zero ကနေ false value အဖြစ်သတ်မှတ်တာကိုတော့ မတွေ့ရပါဘူး။ Default value တစ်ခုကို assign လုပ်ဖို့ 2015 မှာ value ကို assign လုပ်နိုင်ပါတယ်။

For Example:

        
            console.log(3 === 3 && 'a' === 'a');
            // return true
            console.log(3 === 3 && 'cow');
            // return cow
            console.log(3 === 3 && 'cow' && 'chicken');
            // return chicken because all openrands are truthy
            console.log(3 === 3 && false && 'chicken');
            // return false
            3 === 3 && 'cow' && console.log('chicken');
            // return chicken
            console.log(3 === 3 || 'cow' || 'chicken');
            // return true
            console.log(3 === 4 || 'cow' || 'chicken');
            // return cow
            console.log(3 === 4 || false || 0);
            // return 0
